  <script>

    /***************************************************************

     * References to elements

     ***************************************************************/

    const q1Block = document.getElementById("q1");

    const q1NextBtn = document.getElementById("q1-next");

    const q1Checks = document.querySelectorAll('input[name="q1"]');


    const q2Block = document.getElementById("q2");

    const q2NextBtn = document.getElementById("q2-next");

    const q2Checks = document.querySelectorAll('input[name="q2"]');


    const q3q4Block = document.getElementById("q3-q4");

    const q3q4SubmitBtn = document.getElementById("q3q4-submit");

    const q3Radios = document.querySelectorAll('input[name="q3"]');

    const q4Radios = document.querySelectorAll('input[name="q4"]');


    const resultBlock = document.getElementById("result");


    /***************************************************************

     * Utility function: show final result

     ***************************************************************/

    function showResult(cfsText) {

      // Hide all question blocks

      q1Block.classList.add("hidden");

      q2Block.classList.add("hidden");

      q3q4Block.classList.add("hidden");


      // Display the result with summary

      resultBlock.innerHTML = `

        <h2>Your CFS Result</h2>

        <p><strong>${cfsText}</strong></p>

        <p>This classification is based on your self-reported answers. For medical advice and next steps, consult a healthcare professional.</p>

      `;

      resultBlock.classList.remove("hidden");

      window.scrollTo({ top: 0, behavior: 'smooth' });

    }


    /***************************************************************

     * Q1 Logic

     ***************************************************************/

    q1Checks.forEach(checkbox => {

      checkbox.addEventListener("change", () => {

        // The user is allowed to proceed once they've made a selection (or none).

        // We only require them to press "Next" to confirm.

        q1NextBtn.disabled = false;

      });

    });


    q1NextBtn.addEventListener("click", () => {

      const checkedCount = document.querySelectorAll('input[name="q1"]:checked').length;

      if (checkedCount === 4) {

        // All 4 ticked => CFS 7–8

        showResult("CFS 7–8: Severely Frail");

      } else if (checkedCount >= 1 && checkedCount <= 3) {

        // 1–3 ticked => CFS 6

        showResult("CFS 6: Moderately Frail");

      } else {

        // 0 ticked => proceed to Q2

        q1Block.classList.add("hidden");

        q2Block.classList.remove("hidden");

      }

    });


    /***************************************************************

     * Q2 Logic

     ***************************************************************/

    q2NextBtn.addEventListener("click", () => {

      const checkedCountQ2 = document.querySelectorAll('input[name="q2"]:checked').length;

      if (checkedCountQ2 >= 1) {

        // Any ticked => CFS 5

        showResult("CFS 5: Mildly Frail");

      } else {

        // None ticked => proceed to Q3/Q4

        q2Block.classList.add("hidden");

        q3q4Block.classList.remove("hidden");

      }

    });


    /***************************************************************

     * Q3 & Q4 Logic

     ***************************************************************/

    function validateQ3Q4() {

      const q3Selected = document.querySelector('input[name="q3"]:checked');

      const q4Selected = document.querySelector('input[name="q4"]:checked');

      if (q3Selected && q4Selected) {

        q3q4SubmitBtn.disabled = false;

      }

    }


    q3Radios.forEach(radio => radio.addEventListener("change", validateQ3Q4));

    q4Radios.forEach(radio => radio.addEventListener("change", validateQ3Q4));


    q3q4SubmitBtn.addEventListener("click", () => {

      const q3Value = document.querySelector('input[name="q3"]:checked').value;

      const q4Value = document.querySelector('input[name="q4"]:checked').value;


      if (q3Value === "yes" || q4Value === "yes") {

        showResult("CFS 4: Pre-Frail");

      } else {

        showResult("CFS 1–3: Robust");

      }

    });
